055        /**
056         * Make sure that the Quartz scheduler is up and running
057         * 
058         * @throws Exception generic exception
059         */
060        @Test
061        public void testService() throws Exception 
062        {
063                Scheduler scheduler = quartz.getScheduler();
064                assertNotNull(scheduler);
065                assertNotNull(scheduler.getContext());
066        }
067
068        /**
069         * The following test has been updated for Quartz 2.3.0 Adding/removing a
070         * scheduled job
071         * 
072         * @throws Exception generic exception
073         */
074        @Test
075        public void testAddRemoveTrigger() throws Exception 
076        {
077
078                Scheduler scheduler = quartz.getScheduler();
079                TriggerKey triggerKey = TriggerKey.triggerKey("someTrigger", "TURBINE");
080
081                Date date = DateBuilder.dateOf(0, 0, 0, 1, 1, 2099);
082
083                Trigger someDay = TriggerBuilder.newTrigger().withIdentity(triggerKey).forJob("simpleJob", "TURBINE")
084                                .withSchedule(SimpleScheduleBuilder.simpleSchedule().withIntervalInHours(1).repeatForever())
085                                .startAt(date).build();
086
087                scheduler.scheduleJob(someDay);
088
089                Trigger trigger = scheduler.getTrigger(triggerKey);
090                assertNotNull(trigger);
091                scheduler.unscheduleJob(triggerKey);
092                trigger = scheduler.getTrigger(triggerKey);
093                assertNull(trigger);
094
095        }
096
097        /**
098         * Get all scheduled jobs for "TURBINE" to make sure that the registration
099         * worked.
100         * 
101         * @throws Exception generic exception
102         */
103        @Test
104        public void testGetJobs() throws Exception 
105        {
106                Scheduler scheduler = quartz.getScheduler();
107                assertNotNull(scheduler);
108                Set<JobKey> jobNames = scheduler.getJobKeys(GroupMatcher.jobGroupEquals("TURBINE"));
109                assertEquals(2, jobNames.size(), "Expected two registered jobs");
110        }
111
112        /**
113         * Get the job details and job data map of an existing job to make sure that the
114         * XStream configuration works.
115         * 
116         * @throws Exception generic exception
117         */
118        @Test
119        public void testJobDetailMap() throws Exception {
120
121                JobDetail jobDetail = quartz.getScheduler().getJobDetail(JobKey.jobKey("simpleJob", "TURBINE"));
122                assertNotNull(jobDetail);
123                assertEquals("simpleJob", jobDetail.getKey().getName());
124                assertNotNull(jobDetail.getJobDataMap());
125                assertEquals(2, jobDetail.getJobDataMap().size());
126        }
127
128        /**
129         * Make sure the "notSoSimpleJob" is triggered by the CronTrigger.
130         * 
131         * @throws Exception generic exception
132         */
133        @Test
134        public void testGetTriggersOfJob() throws Exception {
135                List<? extends Trigger> triggers = quartz.getScheduler()
136                                .getTriggersOfJob(JobKey.jobKey("notSoSimpleJob", "TURBINE"));
137                assertEquals(1, triggers.size());
138                assertEquals("cronTrigger", ((Trigger) triggers.get(0)).getKey().getName());
139        }
140
141        /**
142         * Make sure that our two registered jobs are executed after one second.
143         * 
144         * @throws Exception generic exception
145         */
146        @Test
147        public void testJobExecution() throws Exception {
148                Thread.sleep(2000);
149                assertTrue(SimpleJob.wasExecuted, "SimpleJob was not executed");
150                assertTrue(NotSoSimpleJob.wasExecuted, "NotSoSimpleJob was not executed");
151                assertTrue(NotSoSimpleJob.wasServiced, "NotSoSimpleJob was not executed");
152                SimpleJob.reset();
153                NotSoSimpleJob.reset();
154                Thread.sleep(2000);
155                assertTrue(SimpleJob.wasExecuted, "SimpleJob was not executed");
156                assertTrue(NotSoSimpleJob.wasExecuted, "NotSoSimpleJob was not executed");
157                assertTrue(NotSoSimpleJob.wasServiced, "NotSoSimpleJob was not serviced");
158        }
